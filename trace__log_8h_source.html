558  inline ChunkTenant *getChunkTenant() {
559  auto shared_index =
560  platform::getCurrentThreadIDCached() % shared_chunks.size();
561 
562  ChunkTenant &cs = (thread_chunk.sentinel)
563  ? thread_chunk
564  : shared_chunks[shared_index];
565 
566  if (!cs.sentinel->acquire()) {
567  resetChunk(cs);
568  return nullptr;
569  }
570  // State is busy
571  if (!cs.chunk || cs.chunk->isFull()) {
572  replaceChunk(cs);
573  if (!cs.chunk) {
574  return nullptr;
575  }
576  }
577 
578  return &cs;
579  }
