479  inline ChunkTenant* getChunkTenant() {
480  auto shared_index = platform::getCurrentThreadIDCached()
481  % shared_chunks.size();
482 
483  ChunkTenant &cs = (thread_chunk.sentinel) ? thread_chunk
484  : shared_chunks[shared_index];
485 
486  if (!cs.sentinel->acquire()) {
487  resetChunk(cs);
488  return nullptr;
489  }
490  // State is busy
491  if (!cs.chunk || cs.chunk->isFull()) {
492  replaceChunk(cs);
493  if (!cs.chunk) {
494  return nullptr;
495  }
496  }
497 
498  return &cs;
499  }
