387  auto shared_index = platform::getCurrentThreadIDCached()
388  % shared_chunks.size();
389 
390  ChunkTenant &cs = (thread_chunk.sentinel) ? thread_chunk
391  : shared_chunks[shared_index];

394  resetChunk(cs);

403  }
