541  inline ChunkTenant* getChunkTenant() {
542  auto shared_index = platform::getCurrentThreadIDCached()
543  % shared_chunks.size();
544 
545  ChunkTenant &cs = (thread_chunk.sentinel) ? thread_chunk
546  : shared_chunks[shared_index];
547 
548  if (!cs.sentinel->acquire()) {
549  resetChunk(cs);
550  return nullptr;
551  }
552  // State is busy
553  if (!cs.chunk || cs.chunk->isFull()) {
554  replaceChunk(cs);
555  if (!cs.chunk) {
556  return nullptr;
557  }
558  }
559 
560  return &cs;
561  }
